  • (prev loc)

Moves to the previous loc in the hierarchy, depth-first. If already at the root, returns nil.

; clojure/zip.clj:215
(defn prev
  [loc]
    (if-let [lloc (left loc)]
      (loop [loc lloc]
        (if-let [child (and (branch? loc) (down loc))]
          (recur (rightmost child))
          loc))
      (up loc)))
