find-doc(clojure.core)

  • (find-doc re-string-or-pattern)

Prints documentation for any var whose documentation or name contains a match for re-string-or-pattern

; clojure/core.clj:3089
(defn find-doc
  [re-string-or-pattern]
    (let [re  (re-pattern re-string-or-pattern)]
      (doseq [ns (all-ns)
              v (sort-by (comp :name meta) (vals (ns-interns ns)))
              :when (and (:doc ^v)
                         (or (re-find (re-matcher re (:doc ^v)))
                             (re-find (re-matcher re (str (:name ^v))))))]
               (print-doc v))))
