zipmap(clojure.core)

  • (zipmap keys vals)

Returns a map with the keys mapped to the corresponding vals.

; core.clj:1669
(defn zipmap
  [keys vals]
    (loop [map {}
           ks (seq keys)
           vs (seq vals)]
      (if (and ks vs)
        (recur (assoc map (first ks) (first vs))
               (next ks)
               (next vs))
        map)))
