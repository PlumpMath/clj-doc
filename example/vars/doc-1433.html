supers(clojure.core)

  • (supers class)

Returns the immediate and indirect superclasses and interfaces of c, if any

; core.clj:3181
(defn supers
  [#^Class class]
  (loop [ret (set (bases class)) cs ret]
    (if (seq cs)
      (let [c (first cs) bs (bases c)]
        (recur (into ret bs) (into (disj cs c) bs)))
      (not-empty ret))))
