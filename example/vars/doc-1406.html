  • (right loc)

Returns the loc of the right sibling of the node at this loc, or nil

; zip.clj:123
(defn right
  [loc]
    (let [[node {l :l  [r & rnext :as rs] :r :as path}] loc]
      (when (and path rs)
        (with-meta [r (assoc path :l (conj l node) :r rnext)] ^loc))))
