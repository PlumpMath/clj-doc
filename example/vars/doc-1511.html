when-first(clojure.core)

Macro

  • (when-first bindings & body)

bindings => x xs

Same as (when (seq xs) (let [x (first xs)] body))

; core.clj:2675
(defmacro when-first
  [bindings & body]
  (assert-args when-first
     (vector? bindings) "a vector for its binding"
     (= 2 (count bindings)) "exactly 2 forms in binding vector")
  (let [[x xs] bindings]
    `(when (seq ~xs)
       (let [~x (first ~xs)]
         ~@body))))
