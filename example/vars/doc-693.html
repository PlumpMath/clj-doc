  • (down loc)

Returns the loc of the leftmost child of the node at this loc, or nil if no children

; clojure/zip.clj:93
(defn down
  [loc]
    (let [[node path] loc
          [c & cnext :as cs] (children loc)]
      (when cs
        (with-meta [c {:l [] 
                       :pnodes (if path (conj (:pnodes path) node) [node]) 
                       :ppath path 
                       :r cnext}] ^loc))))
